iOS开发中获取类名的2种方法：

```
+ (NSString*)getClassName0:(id)obj
{
    const char* className = class_getName([obj class]);
    return [NSString stringWithUTF8String:className];
}

+ (NSString*)getClassName1:(id)obj
{
    return NSStringFromClass([self class]);
}
```
